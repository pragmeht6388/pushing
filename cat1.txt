using ALD.AldFirst.Abstraction.Interfaces.Services;
using ALD.AldFirst.External.Infrastructure.Zoho;
using Microsoft.Extensions.Hosting;
using Serilog;
using System;
using System.Collections.Generic;
using System.Data;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
 
namespace ProspectToDriverConverter
{
   public class ProspectToDriverConvter
    {
        private readonly IRoleService _roleService;
        private readonly IDriverService _driverService;
 
        public ProspectToDriverConvter(IRoleService roleService, IDriverService driverService)
        {
            _roleService = roleService;
            _driverService = driverService;
        }
 
        public async Task ConvertProspectsAsync()
        {
            var prospects = await _roleService.GetAllProspects();
            Console.WriteLine(prospects.Value.Count());
 
 
            foreach(var prospect in prospects.Value)
            {
                try
                {
                    var response = await _driverService.GetDriver(new ALD.AldFirst.Abstraction.Model.Register
                    {
                        EmailAddress = prospect.Email,
 
                    });
                    if (response.IsSuccess)
                    {
                        prospect.RoleId = 1;
                        var updateStatus = await _roleService.UpdateUser(prospect);
                        Log.Logger.Information("Prospect {0} converted to driver: ", prospect.Email);
                        Console.WriteLine("success");
                    }
                    else
                        Console.WriteLine("Failed");
                }
                catch (Exception ex)
                {
                    Log.Logger.Information("Prospect {0} failed convert to driver: ", prospect.Email);
                    Console.WriteLine("Exception : " + ex.Message + " Email: " + prospect.Email);
                }
            }
            Console.ReadLine();
        }
    }
}
